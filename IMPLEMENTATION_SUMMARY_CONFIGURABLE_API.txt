CODE CLÉS:

config.js - Logique principale:
  ─────────────────────────────────────────────────────────
  const determineApiBaseUrl = () => {
    // Priority 1: Injected config
    if (window.ATLASPI_CONFIG?.API_BASE_URL) 
      return window.ATLASPI_CONFIG.API_BASE_URL;
    
    // Priority 2: localStorage
    const stored = localStorage.getItem('atlaspi_api_base_url');
    if (stored) return stored;
    
    // Priority 3: Intelligent fallback
    if (hostname === 'localhost') 
      return "http://localhost:3000";
    
    // Priority 4: Ultimate fallback
    return 'http://localhost:3000';
  };
  ─────────────────────────────────────────────────────────

script.js - Utilisation:
  ─────────────────────────────────────────────────────────
  const API_BASE_URL = window.ATLASPI_CONFIG?.API_BASE_URL 
    || 'http://localhost:3000';
